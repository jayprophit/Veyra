```python
# tax_loss_harvester.py
import pandas as pd
from datetime import datetime, timedelta

class TaxLossHarvester:
    def __init__(self, ghostfolio_api):
        self.gf = ghostfolio_api
        self.tax_year_end = datetime(2026, 4, 5)  # UK tax year
    
    def find_harvest_opportunities(self):
        """Find positions with unrealized losses"""
        positions = self.gf.get_positions()
        opportunities = []
        
        for position in positions:
            current_price = position['market_price']
            avg_cost = position['average_price']
            unrealized_pnl = (current_price - avg_cost) * position['quantity']
            
            if unrealized_pnl < 0:  # Loss position
                days_held = (datetime.now() - position['purchase_date']).days
                
                opportunities.append({
                    'symbol': position['symbol'],
                    'unrealized_loss': abs(unrealized_pnl),
                    'days_held': days_held,
                    'wash_sale_eligible': days_held > 30,  # UK: 30 days
                    'suggested_action': 'SELL_AND_REBUY_SIMILAR'
                })
        
        return sorted(opportunities, key=lambda x: x['unrealized_loss'], reverse=True)
    
    def generate_report(self):
        """Generate tax-loss harvesting recommendations"""
        opportunities = self.find_harvest_opportunities()
        total_harvestable = sum(o['unrealized_loss'] for o in opportunities if o['wash_sale_eligible'])
        
        report = f"""
        TAX-LOSS HARVESTING REPORT
        ==========================
        Tax Year End: {self.tax_year_end.strftime('%Y-%m-%d')}
        Days Remaining: {(self.tax_year_end - datetime.now()).days}
        
        TOTAL HARVESTABLE LOSSES: £{total_harvestable:,.2f}
        Estimated Tax Savings: £{total_harvestable * 0.20:,.2f} (20% CGT rate)
        
        RECOMMENDED ACTIONS:
        """
        
        for opp in opportunities[:5]:  # Top 5 opportunities
            report += f"""
            {opp['symbol']}:
            - Unrealized Loss: £{opp['unrealized_loss']:,.2f}
            - Days Held: {opp['days_held']}
            - Action: {opp['suggested_action']}
            """
        
        return report
```

```python
class FeeAnalyzer:
    def __init__(self, ghostfolio_api):
        self.gf = ghostfolio_api
    
    def analyze_fees(self):
        """Analyze total fees paid across all accounts"""
        accounts = self.gf.get_accounts()
        
        fee_report = {}
        total_fees = 0
        
        for account in accounts:
            transactions = self.gf.get_transactions(account['id'])
            fees = sum(t.get('fee', 0) for t in transactions)
            
            # Estimate platform fees (0.15-0.45% annually for ETFs)
            balance = account['balance']
            platform_fees = balance * 0.003  # Assume 0.3% average
            
            fee_report[account['name']] = {
                'transaction_fees': fees,
                'estimated_platform_fees': platform_fees,
                'total': fees + platform_fees
            }
            total_fees += fees + platform_fees
        
        # Calculate impact on returns
        total_balance = sum(a['balance'] for a in accounts)
        fee_drag = (total_fees / total_balance) * 100
        
        return {
            'by_account': fee_report,
            'total_fees': total_fees,
            'fee_drag_percent': fee_drag,
            'projected_10yr_impact': total_fees * 10  # Simplified
```

```python
# rebalancing_optimizer.py
import numpy as np
from scipy.optimize import minimize

class RebalancingOptimizer:
    def __init__(self, target_allocation):
        """
        target_allocation: dict like {'VTI': 0.6, 'VXUS': 0.3, 'BND': 0.1}
        """
        self.target = target_allocation
    
    def calculate_drift(self, current_holdings, prices):
        """Calculate how far current allocation is from target"""
        total_value = sum(current_holdings[sym] * prices[sym] for sym in current_holdings)
        
        current_allocation = {}
        for sym, qty in current_holdings.items():
            current_allocation[sym] = (qty * prices[sym]) / total_value
        
        drift = {}
        for sym in self.target:
            current_pct = current_allocation.get(sym, 0)
            target_pct = self.target[sym]
            drift[sym] = current_pct - target_pct
        
        return drift, total_value
    
    def generate_rebalancing_trades(self, current_holdings, prices, threshold=0.05):
        """
        Generate trades to rebalance when drift exceeds threshold (5%)
        """
        drift, total_value = self.calculate_drift(current_holdings, prices)
        
        trades = []
        for symbol, drift_pct in drift.items():
            if abs(drift_pct) > threshold:
                # Calculate trade needed
                target_value = total_value * self.target[symbol]
                current_value = current_holdings.get(symbol, 0) * prices[symbol]
                value_diff = target_value - current_value
                
                if value_diff > 0:
                    # Need to buy
                    quantity = value_diff / prices[symbol]
                    trades.append({
                        'symbol': symbol,
                        'action': 'BUY',
                        'quantity': round(quantity, 4),
                        'estimated_value': value_diff,
                        'reason': f"Underweight by {abs(drift_pct):.1%}"
                    })
                else:
                    # Need to sell
                    quantity = abs(value_diff) / prices[symbol]
                    trades.append({
                        'symbol': symbol,
                        'action': 'SELL',
                        'quantity': round(quantity, 4),
                        'estimated_value': abs(value_diff),
                        'reason': f"Overweight by {abs(drift_pct):.1%}"
                    })
        
        return trades

# Example usage
target = {'BTC': 0.40, 'ETH': 0.30, 'SOL': 0.20, 'USDC': 0.10}
optimizer = RebalancingOptimizer(target)

current = {'BTC': 0.3, 'ETH': 0.5, 'SOL': 0.1, 'USDC': 0.1}
prices = {'BTC': 50000, 'ETH': 3000, 'SOL': 150, 'USDC': 1}

trades = optimizer.generate_rebalancing_trades(current, prices)
for trade in trades:
    print(f"{trade['action']} {trade['quantity']} {trade['symbol']} - {trade['reason']}")
```

```python
# goal_planner.py
class GoalBasedPlanner:
    def __init__(self, advisor):
        self.advisor = advisor
        self.goals = []
    
    def add_goal(self, name, target_amount, target_date, priority):
        """Add a financial goal"""
        goal = {
            'name': name,
            'target': target_amount,
            'deadline': target_date,
            'priority': priority,  # 1-5, 5 = highest
            'monthly_contribution_needed': None
        }
        self.goals.append(goal)
        self.calculate_contributions()
    
    def calculate_contributions(self):
        """Calculate monthly contribution needed for each goal"""
        current_net_worth = self.advisor.gf.get_total_value()
        
        for goal in self.goals:
            months_remaining = (goal['deadline'] - datetime.now()).days / 30
            amount_needed = goal['target'] - current_net_worth
            
            if amount_needed > 0:
                # Simple calculation (doesn't account for returns)
                monthly = amount_needed / months_remaining
                goal['monthly_contribution_needed'] = monthly
            else:
                goal['monthly_contribution_needed'] = 0
    
    def generate_plan(self):
        """Generate comprehensive financial plan"""
        total_monthly_needed = sum(g['monthly_contribution_needed'] for g in self.goals)
        available_for_goals = self.advisor.budget.get_monthly_savings_capacity()
        
        plan = f"""
        FINANCIAL GOALS PLAN
        ====================
        
        AVAILABLE MONTHLY: £{available_for_goals:,.2f}
        TOTAL NEEDED: £{total_monthly_needed:,.2f}
        GAP: £{total_monthly_needed - available_for_goals:,.2f}
        
        GOALS:
        """
        
        for goal in sorted(self.goals, key=lambda x: x['priority'], reverse=True):
            plan += f"""
            {goal['name']} (Priority: {goal['priority']}/5)
            Target: £{goal['target']:,.2f} by {goal['deadline'].strftime('%Y-%m-%d')}
            Monthly Contribution Needed: £{goal['monthly_contribution_needed']:,.2f}
            Status: {'ON TRACK' if goal['monthly_contribution_needed'] <= available_for_goals / len(self.goals) else 'NEEDS ATTENTION'}
            """
        
        if total_monthly_needed > available_for_goals:
            plan += f"""
            ⚠️ WARNING: Your goals require £{total_monthly_needed:,.2f}/month,
            but you only have £{available_for_goals:,.2f} available.
            
            Recommendations:
            1. Extend goal deadlines
            2. Reduce target amounts
            3. Increase income
            4. Prioritize highest-priority goals only
            """
        
        return plan

# Usage
planner = GoalBasedPlanner(advisor)
planner.add_goal("Emergency Fund", 15000, datetime(2026, 12, 31), priority=5)
planner.add_goal("House Deposit", 50000, datetime(2028, 6, 30), priority=4)
planner.add_goal("Retirement", 500000, datetime(2045, 1, 1), priority=3)

print(planner.generate_plan())
```

```python
# user_data/strategies/grid_trading.py
from freqtrade.strategy import IStrategy
from pandas import DataFrame
import talib.abstract as ta

class GridTradingStrategy(IStrategy):
    """
    Grid Trading Strategy
    Creates buy/sell orders at regular intervals within a price range
    """
    
    INTERFACE_VERSION = 3
    
    # Strategy parameters
    grid_levels = 10
    grid_lower = 0.90  # 10% below current price
    grid_upper = 1.10  # 10% above current price
    
    # Minimal ROI designed for grid trading
    minimal_roi = {
        "0": 0.01  # 1% profit per grid
    }
    
    # Stoploss
    stoploss = -0.10  # 10% stoploss (grid breaks)
    
    # Trailing stop
    trailing_stop = False
    
    def populate_indicators(self, dataframe: DataFrame, metadata: dict) -> DataFrame:
        """Add indicators"""
        # Current price
        dataframe['price'] = dataframe['close']
        
        # Grid levels
        current_price = dataframe['close'].iloc[-1]
        grid_step = (self.grid_upper - self.grid_lower) * current_price / self.grid_levels
        
        for i in range(self.grid_levels):
            level = self.grid_lower * current_price + (i * grid_step)
            dataframe[f'grid_{i}'] = level
        
        return dataframe
    
    def populate_entry_trend(self, dataframe: DataFrame, metadata: dict) -> DataFrame:
        """Generate buy signals at grid levels"""
        dataframe.loc[:, 'enter_long'] = 0
        
        # Buy when price hits a grid level from above
        for i in range(self.grid_levels):
            grid_col = f'grid_{i}'
            dataframe.loc[
                (dataframe['close'] <= dataframe[grid_col]) &
                (dataframe['close'].shift(1) > dataframe[grid_col].shift(1)),
                'enter_long'
            ] = 1
        
        return dataframe
    
    def populate_exit_trend(self, dataframe: DataFrame, metadata: dict) -> DataFrame:
        """Generate sell signals at next grid level up"""
        dataframe.loc[:, 'exit_long'] = 0
        
        # Sell when price hits next grid level up
        current_price = dataframe['close'].iloc[-1]
        profit_target = current_price * 1.01  # 1% profit
        
        dataframe.loc[
            dataframe['close'] >= profit_target,
            'exit_long'
        ] = 1
        
        return dataframe
```

```python
# crypto_tax_tracker.py
import pandas as pd
from datetime import datetime
import requests

class CryptoTaxTracker:
    def __init__(self):
        self.transactions = []
        self.cost_basis = {}  # symbol -> avg cost
    
    def add_transaction(self, date, symbol, type, quantity, price, fees=0, exchange=''):
        """Add a transaction"""
        tx = {
            'date': pd.to_datetime(date),
            'symbol': symbol,
            'type': type.upper(),  # BUY, SELL, RECEIVE, SEND
            'quantity': float(quantity),
            'price': float(price),
            'fees': float(fees),
            'exchange': exchange,
            'cost_basis': None,
            'gain_loss': None
        }
        
        # Calculate cost basis and gain/loss for sales
        if tx['type'] == 'SELL':
            tx['cost_basis'] = self.cost_basis.get(symbol, price) * quantity
            tx['gain_loss'] = (price * quantity) - tx['cost_basis'] - fees
        elif tx['type'] == 'BUY':
            # Update average cost basis
            current_qty = sum(t['quantity'] for t in self.transactions 
                            if t['symbol'] == symbol and t['type'] == 'BUY')
            current_cost = sum(t['quantity'] * t['price'] for t in self.transactions 
                             if t['symbol'] == symbol and t['type'] == 'BUY')
            
            new_qty = current_qty + quantity
            new_cost = current_cost + (price * quantity) + fees
            self.cost_basis[symbol] = new_cost / new_qty if new_qty > 0 else 0
        
        self.transactions.append(tx)
    
    def import_from_exchange(self, exchange_api):
        """Import transactions from exchange API"""
        # Example for Binance
        trades = exchange_api.get_my_trades()
        for trade in trades:
            self.add_transaction(
                date=trade['time'],
                symbol=trade['symbol'],
                type='BUY' if trade['isBuyer'] else 'SELL',
                quantity=trade['qty'],
                price=trade['price'],
                fees=trade['commission'],
                exchange='Binance'
            )
    
    def calculate_tax_year_summary(self, year):
        """Calculate tax summary for a specific year"""
        year_start = datetime(year, 1, 1)
        year_end = datetime(year, 12, 31)
        
        year_txs = [t for t in self.transactions 
                   if year_start <= t['date'] <= year_end]
        
        # UK Capital Gains Tax calculation
        total_proceeds = sum(t['price'] * t['quantity'] for t in year_txs 
                            if t['type'] == 'SELL')
        total_costs = sum(t.get('cost_basis', 0) for t in year_txs 
                         if t['type'] == 'SELL')
        total_fees = sum(t['fees'] for t in year_txs)
        
        net_gain = total_proceeds - total_costs - total_fees
        
        # UK CGT allowance (2025-2026: £3,000)
        cgt_allowance = 3000
        taxable_gain = max(0, net_gain - cgt_allowance)
        
        # Tax rates (2025-2026)
        # 10% for basic rate taxpayers, 20% for higher rate
        estimated_tax = taxable_gain * 0.20  # Assume higher rate
        
        return {
            'year': year,
            'total_transactions': len(year_txs),
            'total_proceeds': total_proceeds,
            'total_costs': total_costs,
            'total_fees': total_fees,
            'net_gain': net_gain,
            'cgt_allowance_used': min(net_gain, cgt_allowance),
            'taxable_gain': taxable_gain,
            'estimated_tax': estimated_tax
        }
    
    def generate_tax_report(self, year):
        """Generate detailed tax report"""
        summary = self.calculate_tax_year_summary(year)
        
        report = f"""
        CRYPTO TAX REPORT - {year}
        =========================
        
        SUMMARY
        -------
        Total Transactions: {summary['total_transactions']}
        Total Proceeds: £{summary['total_proceeds']:,.2f}
        Total Costs: £{summary['total_costs']:,.2f}
        Total Fees: £{summary['total_fees']:,.2f}
        
        NET GAIN: £{summary['net_gain']:,.2f}
        
        TAX CALCULATION
        ---------------
        CGT Allowance Used: £{summary['cgt_allowance_used']:,.2f}
        Taxable Gain: £{summary['taxable_gain']:,.2f}
        Estimated Tax Due: £{summary['estimated_tax']:,.2f}
        
        TRANSACTIONS DETAIL
        -------------------
        """
        
        year_txs = [t for t in self.transactions 
                   if datetime(year, 1, 1) <= t['date'] <= datetime(year, 12, 31)]
        
        for tx in sorted(year_txs, key=lambda x: x['date']):
            if tx['type'] == 'SELL':
                report += f"""
                {tx['date'].strftime('%Y-%m-%d')} - SELL {tx['symbol']}
                Quantity: {tx['quantity']:.6f}
                Price: £{tx['price']:,.2f}
                Proceeds: £{tx['price'] * tx['quantity']:,.2f}
                Cost Basis: £{tx.get('cost_basis', 0):,.2f}
                Fees: £{tx['fees']:.2f}
                GAIN/(LOSS): £{tx.get('gain_loss', 0):,.2f}
                """
        
        return report
```
