```python
# risk_metrics.py
import numpy as np
import pandas as pd

class RiskMetrics:
    def __init__(self, returns, risk_free_rate=0.02):
        self.returns = np.array(returns)
        self.risk_free_rate = risk_free_rate
    
    def sharpe_ratio(self):
        """Sharpe Ratio = (Return - Risk Free) / Std Dev"""
        excess_returns = self.returns - self.risk_free_rate / 252  # Daily
        if np.std(self.returns) == 0:
            return 0
        return np.mean(excess_returns) / np.std(self.returns) * np.sqrt(252)
    
    def max_drawdown(self):
        """Maximum peak-to-trough decline"""
        cumulative = np.cumprod(1 + self.returns)
        running_max = np.maximum.accumulate(cumulative)
        drawdown = (cumulative - running_max) / running_max
        return np.min(drawdown)
    
    def win_rate(self):
        """Percentage of positive returns"""
        return np.mean(self.returns > 0)
    
    def profit_factor(self):
        """Gross Profit / Gross Loss"""
        gains = np.sum(self.returns[self.returns > 0])
        losses = np.abs(np.sum(self.returns[self.returns < 0]))
        return gains / losses if losses != 0 else float('inf')
    
    def can_advance_tier(self, current_tier):
        """Check if strategy qualifies for next tier"""
        metrics = {
            'sharpe': self.sharpe_ratio(),
            'max_drawdown': self.max_drawdown(),
            'win_rate': self.win_rate(),
            'profit_factor': self.profit_factor()
        }
        
        # Tier 1 → 2 requirements
        if current_tier == 1:
            return (
                metrics['sharpe'] > 1.0 and
                metrics['max_drawdown'] > -0.15 and
                metrics['win_rate'] > 0.55
            )
        
        # Tier 2 → 3 requirements
        if current_tier == 2:
            return (
                metrics['sharpe'] > 1.2 and
                metrics['max_drawdown'] > -0.12 and
                metrics['win_rate'] > 0.60
            )
        
        return False
```
