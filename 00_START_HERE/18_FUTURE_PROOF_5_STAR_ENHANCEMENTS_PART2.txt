    def execute_intent(self, intent: str, context: Dict) -> Dict:
        """
        Convert natural language intent to transaction
        
        Examples:
        - "Move my ETH to the highest yield protocol"
        - "Rebalance my portfolio to 60/40 stocks/bonds"
        - "Take profits on my BTC position"
        """
        # 1. Parse intent using LLM
        parsed_intent = self._parse_intent(intent)
        
        # 2. Validate against guardrails
        if not self._validate_intent(parsed_intent):
            return {
                'status': 'REJECTED',
                'reason': 'Intent violates guardrails',
                'intent': parsed_intent
            }
        
        # 3. Generate transaction plan
        tx_plan = self._generate_transaction_plan(parsed_intent, context)
        
        # 4. Human approval for high-value (optional based on settings)
        if tx_plan['total_value'] > self.config.get('approval_threshold', 100):
            return {
                'status': 'PENDING_APPROVAL',
                'plan': tx_plan,
                'approval_required_from': self.owner_address
            }
        
        # 5. Execute via guardrails contract
        return self._execute_via_guardrails(tx_plan)
    
    def _parse_intent(self, intent: str) -> Dict:
        """Use LLM to parse natural language intent"""
        # Example using DeepSeek API
        prompt = f"""
        Parse this financial intent into structured data:
        Intent: "{intent}"
        
        Extract:
        1. Action type (move, rebalance, buy, sell, stake, etc.)
        2. Assets involved
        3. Target/destination
        4. Constraints or conditions
        5. Amount or percentage
        
        Return as JSON.
        """
        
        # Call LLM API
        response = call_llm_api(prompt)
        return json.loads(response)
    
    def _validate_intent(self, parsed_intent: Dict) -> bool:
        """Check if parsed intent is within guardrails"""
        guardrails = self._load_guardrails()
        
        # Check daily spend limit
        today_spend = self._get_today_spend()
        if today_spend + parsed_intent.get('amount', 0) > guardrails['daily_limit']:
            return False
        
        # Check single transaction limit
        if parsed_intent.get('amount', 0) > guardrails['tx_limit']:
            return False
        
        # Check allowed protocols
        if parsed_intent.get('target_protocol') not in guardrails['allowed_protocols']:
            return False
        
        # Check restricted assets
        if parsed_intent.get('asset') in guardrails['restricted_assets']:
            return False
        
        return True
    
    def _execute_via_guardrails(self, tx_plan: Dict) -> Dict:
        """Execute transaction through guardrails smart contract"""
        # Build transaction
        tx = {
            'to': tx_plan['target_contract'],
            'value': tx_plan['value'],
            'data': tx_plan['calldata'],
            'gas': tx_plan['estimated_gas'],
            'gasPrice': tx_plan['gas_price']
        }
        
        # Sign with AI agent key
        signed_tx = self.safe_account.sign_transaction(tx, signer=self.ai_agent_address)
        
        # Send to guardrails contract for validation and execution
        return self.guardrails.executeTransaction(
            tx['to'], tx['value'], tx['data']
        )

# Verifiable AI - On-chain decision logging
class VerifiableAI:
    """
    Every AI decision is logged on-chain for auditability
    """
    
    def __init__(self):
        self.attestation_contract = "0x..."  # Smart contract for storing decisions
    
    def log_decision(self, decision: Dict, reasoning: str) -> str:
        """
        Log AI decision to blockchain for permanent record
        Returns attestation hash
        """
        decision_hash = Web3.keccak(text=json.dumps(decision, sort_keys=True))
        reasoning_hash = Web3.keccak(text=reasoning)
        
        # Store on-chain
        tx = self._store_attestation({
            'decision_hash': decision_hash.hex(),
            'reasoning_hash': reasoning_hash.hex(),
            'timestamp': int(time.time()),
            'ai_version': self.get_ai_version(),
            'model_hash': self.get_model_hash()
        })
        
        return tx['transactionHash'].hex()
    
    def verify_decision(self, attestation_hash: str) -> Dict:
        """
        Verify a past AI decision using on-chain record
        """
        # Retrieve from blockchain
        record = self._get_attestation(attestation_hash)
        
        # Verify hashes match stored decision
        return {
            'verified': True,
            'timestamp': record['timestamp'],
            'ai_version': record['ai_version'],
            'decision_hash': record['decision_hash']
        }
