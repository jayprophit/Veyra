```python
# ai_advisor_simple.py - Quick start version
import os
from openai import OpenAI

class SimpleAIAdvisor:
    def __init__(self):
        # Use DeepSeek for cost savings ($0.07/1M tokens)
        self.client = OpenAI(
            api_key=os.getenv('DEEPSEEK_API_KEY'),
            base_url="https://api.deepseek.com"
        )
    
    def analyze_portfolio(self, portfolio_data):
        prompt = f"""
        Analyze this portfolio and provide 3 specific optimization recommendations:
        
        Portfolio: {portfolio_data}
        
        Consider:
        1. Asset allocation balance
        2. Risk concentration
        3. Tax efficiency
        4. Fee optimization
        
        Provide actionable, specific advice.
        """
        
        response = self.client.chat.completions.create(
            model="deepseek-chat",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.3,
            max_tokens=1000
        )
        
        return response.choices[0].message.content

# Usage
advisor = SimpleAIAdvisor()
portfolio = {
    "total_value": 50000,
    "assets": {
        "BTC": {"value": 10000, "allocation": 0.20},
        "ETH": {"value": 8000, "allocation": 0.16},
        "VWRP": {"value": 20000, "allocation": 0.40},
        "Cash": {"value": 12000, "allocation": 0.24}
    },
    "monthly_contribution": 1000
}

print(advisor.analyze_portfolio(portfolio))
```

```python
                (dataframe['close'] < dataframe['bb_lower'])  # Below lower band
            ),
            'enter_long'
        ] = 1
        
        return dataframe
    
    def populate_exit_trend(self, dataframe: DataFrame, metadata: dict) -> DataFrame:
        dataframe.loc[
            (
                (dataframe['rsi'] > self.sell_rsi) |  # Overbought
                (dataframe['close'] > dataframe['bb_upper'])  # Above upper band
            ),
            'exit_long'
        ] = 1
        
        return dataframe
```

```python
# retirement_simulator.py
import numpy as np
import pandas as pd

class RetirementSimulator:
    def __init__(self, current_age, retirement_age, current_savings, 
                 annual_contribution, target_amount):
        self.current_age = current_age
        self.retirement_age = retirement_age
        self.current_savings = current_savings
        self.annual_contribution = annual_contribution
        self.target_amount = target_amount
        self.years_to_retirement = retirement_age - current_age
    
    def monte_carlo_simulation(self, n_simulations=10000, 
                               expected_return=0.07, 
                               volatility=0.15):
        """Run Monte Carlo simulation for retirement projections"""
        results = []
        
        for _ in range(n_simulations):
            portfolio_value = self.current_savings
            yearly_values = [portfolio_value]
            
            for year in range(self.years_to_retirement):
                # Random return based on normal distribution
                annual_return = np.random.normal(expected_return, volatility)
                
                portfolio_value = portfolio_value * (1 + annual_return)
                portfolio_value += self.annual_contribution
                yearly_values.append(portfolio_value)
            
            results.append(yearly_values)
        
        # Calculate percentiles
        results_array = np.array(results)
        percentiles = {
            '10th': np.percentile(results_array[:, -1], 10),
            '25th': np.percentile(results_array[:, -1], 25),
            '50th': np.percentile(results_array[:, -1], 50),
            '75th': np.percentile(results_array[:, -1], 75),
            '90th': np.percentile(results_array[:, -1], 90)
        }
        
        # Calculate success probability
        successes = sum(1 for r in results if r[-1] >= self.target_amount)
        success_rate = successes / n_simulations
        
        return {
            'percentiles': percentiles,
            'success_rate': success_rate,
            'median_final_value': percentiles['50th'],
            'all_simulations': results
        }
    
    def safe_withdrawal_rate(self, retirement_balance, years_in_retirement=30):
        """Calculate safe withdrawal rate using 4% rule variations"""
        
        # Traditional 4% rule
        traditional_4pct = retirement_balance * 0.04
        
        # Adjusted for current market conditions (3.5-3.75%)
        conservative_35pct = retirement_balance * 0.035
        
        # Dynamic withdrawal based on market performance
        dynamic_withdrawal = {
            'good_market': retirement_balance * 0.05,
            'average_market': retirement_balance * 0.04,
            'poor_market': retirement_balance * 0.03
        }
        
        return {
            'traditional_4_percent': traditional_4pct,
            'conservative_3_5_percent': conservative_35pct,
            'dynamic_strategy': dynamic_withdrawal
        }
```
