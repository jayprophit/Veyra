```python
# src/control_layer/risk_manager.py
class RiskManager:
    def __init__(self, config: Dict):
        self.max_position_size = config.get('max_position_size', 0.05)  # 5%
        self.max_daily_loss = config.get('max_daily_loss', 0.02)  # 2%
        self.max_portfolio_heat = config.get('max_portfolio_heat', 0.20)  # 20%
        self.current_daily_loss = 0.0
    
    def check_trade(self, trade: Dict, portfolio: Dict) -> bool:
        # Check position size
        trade_value = trade['amount'] * trade['price']
        portfolio_value = portfolio['total_value']
        position_pct = trade_value / portfolio_value
        
        if position_pct > self.max_position_size:
            print(f"REJECTED: Position size {position_pct:.2%} exceeds max {self.max_position_size:.2%}")
            return False
        
        # Check daily loss limit
        if trade['side'] == 'sell' and trade.get('unrealized_pnl', 0) < 0:
            potential_loss = abs(trade['unrealized_pnl'])
            if self.current_daily_loss + potential_loss > self.max_daily_loss * portfolio_value:
                print(f"REJECTED: Daily loss limit would be exceeded")
                return False
        
        return True
    
    def update_daily_loss(self, pnl: float):
        self.current_daily_loss += pnl
        if self.current_daily_loss > 0:
            self.current_daily_loss = 0  # Reset on profit
    
    def reset_daily_loss(self):
        self.current_daily_loss = 0.0
```

```python
import ccxt
import pandas as pd
from typing import Dict, Optional

class MarketDataFeed:
    def __init__(self, exchange_id: str = 'binance'):
        self.exchange = getattr(ccxt, exchange_id)()
    
    def fetch_ohlcv(self, symbol: str, timeframe: str = '1h', 
                   limit: int = 100) -> pd.DataFrame:
        """Fetch OHLCV candles"""
        data = self.exchange.fetch_ohlcv(symbol, timeframe, limit=limit)
        df = pd.DataFrame(data, columns=['timestamp', 'open', 'high', 
                                        'low', 'close', 'volume'])
        df['timestamp'] = pd.to_datetime(df['timestamp'], unit='ms')
        return df
    
    def fetch_ticker(self, symbol: str) -> Dict:
        """Fetch current price and stats"""
        return self.exchange.fetch_ticker(symbol)
    
    def fetch_balance(self, api_key: str, secret: str) -> Dict:
        """Fetch account balance (paper trading)"""
        self.exchange.apiKey = api_key
        self.exchange.secret = secret
        return self.exchange.fetch_balance()
```

```python
# src/execution_layer/exchange_connector.py
class PaperTradingConnector:
    def __init__(self, initial_balance: float = 10000):
        self.balance = {'USDT': initial_balance}
        self.positions = {}
        self.trades = []
    
    def place_order(self, symbol: str, side: str, amount: float, 
                   price: float, order_type: str = 'market') -> Dict:
        """Simulate order execution"""
        # Get current market price (simulated with slight slippage)
        execution_price = price * (1 + np.random.normal(0, 0.001))
        
        # Calculate cost/proceeds
        if side == 'buy':
            cost = amount * execution_price
            if cost > self.balance.get('USDT', 0):
                return {'status': 'rejected', 'reason': 'insufficient_balance'}
            
            self.balance['USDT'] -= cost
            self.positions[symbol] = self.positions.get(symbol, 0) + amount
        else:  # sell
            if amount > self.positions.get(symbol, 0):
                return {'status': 'rejected', 'reason': 'insufficient_position'}
            
            proceeds = amount * execution_price
            self.balance['USDT'] = self.balance.get('USDT', 0) + proceeds
            self.positions[symbol] -= amount
        
        trade = {
            'id': len(self.trades) + 1,
            'symbol': symbol,
            'side': side,
            'amount': amount,
            'price': execution_price,
            'timestamp': datetime.now(),
            'status': 'filled'
        }
        self.trades.append(trade)
        return trade
    
    def get_portfolio_value(self, current_prices: Dict) -> float:
        """Calculate total portfolio value"""
        cash = self.balance.get('USDT', 0)
        positions_value = sum(
            self.positions[sym] * current_prices.get(sym, 0) 
            for sym in self.positions
        )
        return cash + positions_value
```

```python
# src/reasoning_layer/strategies/dca_strategy.py
class DCAStrategy:
    def __init__(self, symbol: str, amount: float, frequency: str):
        self.symbol = symbol
        self.amount = amount  # £ amount to buy
        self.frequency = frequency  # 'daily', 'weekly'
        self.last_buy = None
    
    def should_buy(self, current_time: datetime) -> bool:
        if self.last_buy is None:
            return True
        
        if self.frequency == 'daily':
            return (current_time - self.last_buy).days >= 1
        elif self.frequency == 'weekly':
            return (current_time - self.last_buy).days >= 7
        return False
    
    def generate_signal(self, market_data: Dict) -> Dict:
        current_time = datetime.now()
        if self.should_buy(current_time):
            return {
                'action': 'buy',
                'symbol': self.symbol,
                'amount': self.amount,
                'confidence': 1.0,  # DCA always executes
                'reason': f"DCA schedule: {self.frequency} purchase"
            }
        return {'action': 'hold'}
```

**Strategy B: Moving Average Crossover**
```python
class MACrossoverStrategy:
    def __init__(self, fast_period: int = 10, slow_period: int = 50):
        self.fast_period = fast_period
        self.slow_period = slow_period
    
    def generate_signal(self, df: pd.DataFrame) -> Dict:
        df['fast_ma'] = df['close'].rolling(self.fast_period).mean()
        df['slow_ma'] = df['close'].rolling(self.slow_period).mean()
        
        current = df.iloc[-1]
        previous = df.iloc[-2]
        
        # Golden cross: fast crosses above slow
        if previous['fast_ma'] <= previous['slow_ma'] and \
           current['fast_ma'] > current['slow_ma']:
            return {'action': 'buy', 'confidence': 0.7, 'reason': 'Golden cross'}
        
        # Death cross: fast crosses below slow
        elif previous['fast_ma'] >= previous['slow_ma'] and \
             current['fast_ma'] < current['slow_ma']:
            return {'action': 'sell', 'confidence': 0.7, 'reason': 'Death cross'}
        
        return {'action': 'hold'}
```

```python
# src/orchestrator.py
class TradingOrchestrator:
    def __init__(self, config: Dict):
        self.data_feed = MarketDataFeed()
        self.risk_manager = RiskManager(config['risk'])
        self.execution = PaperTradingConnector(config['initial_balance'])
        self.strategies = []
        self.running = False
    
    def register_strategy(self, strategy):
        self.strategies.append(strategy)
    
    def run(self):
        self.running = True
        print("Orchestrator started...")
        
        while self.running:
            try:
                # 1. Fetch market data
                for strategy in self.strategies:
                    symbol = strategy.symbol
                    market_data = self.data_feed.fetch_ohlcv(symbol, '1h', 100)
                    current_price = market_data['close'].iloc[-1]
                    
                    # 2. Generate signals
                    signal = strategy.generate_signal(market_data)
                    
                    # 3. Check risk limits
                    portfolio = self.execution.get_portfolio_value({symbol: current_price})
                    trade = {
                        'symbol': signal['symbol'],
                        'side': signal['action'],
                        'amount': signal.get('amount', 0),
                        'price': current_price
                    }
                    
                    if self.risk_manager.check_trade(trade, {'total_value': portfolio}):
                        # 4. Execute trade
                        result = self.execution.place_order(**trade)
                        print(f"Trade executed: {result}")
                        
                        # 5. Log and audit
                        self.log_trade(result)
                    else:
                        print(f"Trade rejected by risk manager: {trade}")
                
                # Sleep until next cycle
                time.sleep(3600)  # Check every hour
                
            except Exception as e:
                print(f"Error in main loop: {e}")
                time.sleep(60)  # Wait before retrying
    
    def stop(self):
        self.running = False
        print("Orchestrator stopped")
```

WHY: Coordinates all components into unified system
WHERE: Main application entry point
THRESHOLD: Must run for 24+ hours without crashing

```python
# Additional safety for live trading
class LiveSafetyWrapper:
    def __init__(self, connector, max_order_size: float = 100):
        self.connector = connector
        self.max_order_size = max_order_size  # £100 max per order (Tier 1)
        self.daily_volume = 0
        self.last_reset = datetime.now()
    
    def place_order(self, **kwargs):
        # Reset daily volume counter
        if (datetime.now() - self.last_reset).days >= 1:
            self.daily_volume = 0
            self.last_reset = datetime.now()
        
        # Check order size
        order_value = kwargs['amount'] * kwargs.get('price', 0)
        if order_value > self.max_order_size:
            print(f"REJECTED: Order {order_value} exceeds max {self.max_order_size}")
            return {'status': 'rejected', 'reason': 'size_limit'}
        
        # Check daily volume limit
        if self.daily_volume + order_value > self.max_order_size * 5:  # 5 trades/day max
            print("REJECTED: Daily volume limit reached")
            return {'status': 'rejected', 'reason': 'daily_limit'}
        
        # Execute
        result = self.connector.place_order(**kwargs)
        if result.get('status') == 'filled':
            self.daily_volume += order_value
        
        return result
```

```python
class SentinelAgent:
    """Monitors all other agents for anomalous behavior"""
    
    def __init__(self):
        self.baseline_metrics = {}
        self.anomaly_threshold = 3.0  # Standard deviations
    
    def monitor_trading_agent(self, agent_logs: List[Dict]):
        # Analyze patterns in agent decisions
        recent_trades = agent_logs[-100:]
        
        # Detect anomalies:
        # 1. Unusual trade frequency
        trade_frequency = len(recent_trades) / 24  # per hour
        if trade_frequency > self.baseline_metrics.get('avg_freq', 2) * 3:
            self.alert("ABNORMAL: Trading frequency 3x above normal")
            return False
        
        # 2. Unusual trade sizes
        avg_size = np.mean([t['size'] for t in recent_trades])
        last_trade_size = recent_trades[-1]['size']
        if last_trade_size > avg_size * 5:
            self.alert("ABNORMAL: Trade size 5x above average")
            return False
        
        # 3. Unusual asset selection
        recent_symbols = [t['symbol'] for t in recent_trades[-10:]]
        if len(set(recent_symbols)) > 5:  # Trading too many different assets
            self.alert("ABNORMAL: Diversification spike - possible confusion")
            return False
        
        return True
    
    def emergency_stop(self):
        """Revoke all API keys, pause all trading"""
        for exchange in self.connected_exchanges:
            exchange.cancel_all_orders()
            self.revoke_api_key(exchange.api_key)
        self.send_alert("EMERGENCY STOP ACTIVATED", priority='CRITICAL')
```
